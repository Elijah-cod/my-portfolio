async def resolve_workspace_query(user, payload):
    if not permissions.can_search(user, payload.workspace_id):
        raise ForbiddenError()

    matches = retriever.rank(
        query=payload.query,
        vectors="workspace_index",
        graph_scope=payload.workspace_id,
    )

    context = graph.expand(matches[:3], depth=2)
    return llm.answer(query=payload.query, context=context)

The key decision is where responsibility lives. Permission checks, retrieval scoping, and context assembly happen before the model call, because product safety belongs earlier in the path.
